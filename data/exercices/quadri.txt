// BKRESET.SPLIT:quadrilater.h
#include <iostream>
#include <string>

using namespace std;

class Quadrilateral {
  public:
    void properties(){cout << "Quadrilateral: Je suis une figure à quatre côté quelconque." << endl;}
};

class Trapezoid: public Quadrilateral {
  public:
    void properties(){cout << "Trapezoid: J'ai deux côtés parallèles uniquement." << endl;}
};

class Rectangle: public Quadrilateral {
  public:
    void properties(){cout << "Rectangle: J'ai mes côtés parallèles deux à deux." << endl;}
};

class Square: public Rectangle {
  public:
    void properties(){cout << "Square: J'ai mes côtés parallèles deux à deux et de même longueur." << endl;}
};
// BKRESET.SPLIT:quadrilater.cpp
#include "quadrilater.h"

int main() {
  Quadrilateral().properties();
  Trapezoid().properties();
  Rectangle().properties();
  Square().properties();

  return 0;
}
